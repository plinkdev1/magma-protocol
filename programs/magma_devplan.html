// 06Anchor Program Specifications
magma_registryNarrative Registry — cNFT Mint + Scoring Gate
submitNarrative(thesis_cid, title, category)
Creates NarrativeAccount PDA. Validates originality hash. Sets status = Pending. Emits NarrativeSubmitted event.
// signer: creator wallet · fee: 0.001 SOL to treasury
mintNarrativeNFT(narrative_pda)
CPI to Metaplex Bubblegum MintToCollectionV1. Stores cNFT leaf index in NarrativeAccount. Sets status = Active. MWA signs on mobile.
// constraint: narrative.status == Pending · signer: creator
openScoring(narrative_pda)
Validates score ≥ 70 precondition. Opens backing window. Sets backing_open = true, cycle_start = Clock::get().
// #[only_agent] — restricted to oracle_signer pubkey
graduateNarrative(narrative_pda)
Validates score ≥ 85 AND sol_backed ≥ 1 SOL. Sets status = Graduated. Emits Graduated event for ORIGIN bridge listener.
// #[only_agent] · emits Wormhole cross-chain event
magma_vaultDeFi Treasury — Stake + Yield + Payout
stakeOnNarrative(narrative_pda, amount_lamports)
Accepts SOL. Creates BackerPosition PDA (backer, narrative, amount, timestamp). If Clock::unix_timestamp - cycle_start < 172800 (48h), sets early_backer = true → 1.5× multiplier on payout. Immediately deploys to DeFi vault via CPI.
// #[payable("SOL")] · constraint: narrative.status == Active
deployToDeFi(vault_pda, allocation)
Splits vault SOL: 40% → Meteora DLMM add_liquidity CPI, 35% → Kamino deposit CPI, 25% → Save.Finance deposit_reserve_liquidity CPI. Stores position receipts.
// #[only_agent] · called after each stakeOnNarrative
closeCycle(narrative_pda)
Redeems all DeFi positions. Calculates total yield. Distributes: 60% → creator, 30% → backers (pro-rata + early multiplier), 10% → $MAGMA stakers pool. Burns 20% of $MAGMA for slashed narratives (score < 45).
// #[only_agent] · weekly cron · validates cycle_end
magma_scoreScore Oracle — Pyth-Validated Daily Writes
updateScore(narrative_pda, score, component_scores)
Validates Pyth SOL/USD price account freshness (max 60s stale). Writes composite score daily. Score = (factual×0.30) + (engagement×0.30) + (originality×0.20) + (community×0.20). Emits ScoreUpdated event.
// #[only_agent("A-06 Score Oracle")] · Pyth staleness check required
recordCommunityVote(narrative_pda, vote_direction)
Records positive/negative vote from $MAGMA holder (min 100 $MAGMA). Contributes to community_vote component. One vote per wallet per narrative.
// constraint: voter.magma_balance ≥ 100 · no double-vote
magma_token$MAGMA SPL Token-2022 + TransferHook
execute (TransferHook interface)
Called by Token-2022 program on every $MAGMA transfer. If destination is a narrative backing account, automatically creates BackerPosition PDA entry. Transfer + stake = one transaction. No double-approve required.
// interface: spl_transfer_hook_interface::execute · one tx = transfer + stake
initialize_extra_account_meta_list()
Required by TransferHook interface. Sets up ExtraAccountMetaList for the hook to reference magma_vault program and backing PDAs on every transfer.
// called once at token init · required by Token-2022 spec
// 07MWA Integration — Critical Code Patterns
This is the most important section for Monolith eligibility

Every single Solana transaction in MAGMA must go through Mobile Wallet Adapter. Using WalletConnect, a WebView, or any browser-based signing will be caught by judges reviewing source code and will result in disqualification or very low scores.

// useAuthorization.ts — wallet connection hook import { transact, Web3MobileWallet } from '@solana-mobile/mobile-wallet-adapter-protocol-web3js'; import { PublicKey } from '@solana/web3.js'; const APP_IDENTITY = { name: 'MAGMA', uri: 'https://magmaprotocol.xyz', icon: 'favicon.ico', }; export function useAuthorization() { const [account, setAccount] = useState<PublicKey | null>(null); const connect = useCallback(async () => { await transact(async (wallet: Web3MobileWallet) => { const auth = await wallet.authorize({ cluster: 'mainnet-beta', identity: APP_IDENTITY, }); setAccount(new PublicKey(auth.accounts[0].address)); }); }, []); return { account, connect }; }
// useMagmaTransactions.ts — MWA transaction signing import { transact } from '@solana-mobile/mobile-wallet-adapter-protocol-web3js'; import * as Haptics from 'expo-haptics'; export function useMintNarrative() { return useCallback(async (kitCid: string, thesisCid: string) => { return await transact(async (wallet) => { // Build mint instruction via Anchor client const tx = await buildMintNarrativeTx(kitCid, thesisCid); // Sign natively — no browser, no WebView const [signed] = await wallet.signTransactions({ transactions: [tx] }); // Haptic feedback confirms signing await Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium); return await connection.sendRawTransaction(signed.serialize()); }); }, []); }
// usePythPriceFeed.ts — 400ms SOL/USD oracle const SOL_USD_FEED = 'H6ARHf6YXhGYeQfUzQNGk6rDNnLBQKrenN712K4AQJEG'; export function usePythPriceFeed() { const [price, setPrice] = useState<number | null>(null); useEffect(() => { const pythConn = new PythConnection(connection, PYTH_PROGRAM_KEY); const feedKey = new PublicKey(SOL_USD_FEED); const interval = setInterval(async () => { const priceData = await pythConn.getLatestPriceFeeds([feedKey]); if (priceData[0]?.price) setPrice(priceData[0].price.price); }, 400); return () => clearInterval(interval); }, []); return price; }
